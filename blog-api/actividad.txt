USE blogdb;

CREATE TABLE favoritos (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    conteo_favoritos BIGINT DEFAULT 0 NOT NULL,
    id_usuario BIGINT,
    id_comentario BIGINT,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id) ON DELETE CASCADE,
    FOREIGN KEY (id_comentario) REFERENCES comentarios(id) ON DELETE CASCADE
 );

 ALTER TABLE favoritos RENAME TO favoritos_comentarios;

CREATE TABLE favoritos_blogs (
 	id BIGINT AUTO_INCREMENT PRIMARY KEY,
    favorito BOOLEAN,
    usuario_id BIGINT,
    blog_id BIGINT,
    FOREIGN KEY(usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY(blog_id) REFERENCES blogs(id) ON DELETE CASCADE ON UPDATE CASCADE
 );
